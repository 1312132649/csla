  public class MySafeDataReader : SafeDataReader
  {
    public MySafeDataReader(IDataReader dataReader ) : base(dataReader)
    {
    }
 
    public System.TimeSpan GetTime(string field )
    {
      // implementation goes here......
    }
  }
