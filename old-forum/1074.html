public C Find(Predicate<C> match)
{
      if (match == null)
      {
            throw new ArgumentNullException("match");
      }
      for (int index = 0; index < base.Count; index++)
      {
            if (match(base[index]))
            {
                  return base[index];
            }
      }
      return default(C);
}
